function trocarAba(id, btn) { document.querySelectorAll('.aba').forEach(a => a.classList.remove('ativa')); document.getElementById(id).classList.add('ativa'); document.querySelectorAll('.menu button').forEach(b => b.classList.remove('ativo')); btn.classList.add('ativo'); } function carregarProdutos() { let select = document.getElementById("produto"); select.innerHTML = ""; for (let tipo in produtos) { let group = document.createElement('optgroup'); group.label = tipo; produtos[tipo].forEach(p => { let opt = document.createElement("option"); opt.text = p.desc; opt.value = p.tempo; group.appendChild(opt); }); select.appendChild(group); } } function adicionarProduto() { let select = document.getElementById("produto"); let qtd = parseInt(document.getElementById("quantidade").value); let tempoTotal = parseInt(select.value) * qtd; itensSelecionados.push({ desc: `${select.options[select.selectedIndex].text} (Qtd: ${qtd})`, tempo: tempoTotal }); renderizarListaProdutos(); } function renderizarListaProdutos() { let totalMinutos = itensSelecionados.reduce((acc, i) => acc + i.tempo, 0); document.getElementById("horasNecessarias").innerText = Math.ceil(totalMinutos / 60); document.getElementById("lista-produtos").innerHTML = itensSelecionados.map((i, idx) => `
${i.desc}
`).join(''); } function removerProduto(idx) { itensSelecionados.splice(idx, 1); renderizarListaProdutos(); } function iniciarAgendamento() { if(prestadores.length === 0) { alert("Cadastre um prestador primeiro!"); return; } document.getElementById('step-cep').style.display = 'none'; document.getElementById('step-agendamento').style.display = 'block'; } function renderizarAgendaAgendamento() { let idx = document.getElementById("selectPrestador").value; let c = document.getElementById("agenda-agendamento"); c.innerHTML = ""; if(idx === "") return; prestadores[idx].agenda.forEach(hora => { let b = document.createElement("div"); b.className = "cell"; b.innerText = hora; b.onclick = () => b.classList.toggle("selecionado"); c.appendChild(b); }); } function renderizarListaPrestadores() { let l = document.getElementById("lista-prestadores"); l.innerHTML = prestadores.map((p, i) => `
${p.nome}
`).join(''); } function excluirPrestador(i) { prestadores.splice(i, 1); renderizarListaPrestadores(); atualizarSelectPrestadores(); } function atualizarSelectPrestadores() { let sel = document.getElementById("selectPrestador"); sel.innerHTML = ''; prestadores.forEach((p, i) => { sel.innerHTML += ``; }); } function mudarEtapa(dir) { document.getElementById(`etapa-${etapa}`).style.display = "none"; etapa += dir; if(etapa < 1) etapa = 1; if(etapa > 4) etapa = 4; document.getElementById(`etapa-${etapa}`).style.display = "block"; document.getElementById("etapaAtual").innerText = etapa; document.getElementById("btnProximo").style.display = etapa === 4 ? "none" : "inline"; document.getElementById("btnSalvar").style.display = etapa === 4 ? "inline" : "none"; } function resetCadastro() { document.getElementById("lista-prestadores-view").style.display = "block"; document.getElementById("cadastro-flow").style.display = "none"; etapa = 1; } function resetarTelaParaInicio() { location.reload(); } function concluirAgendamento() { let slotsMax = parseInt(document.getElementById("horasNecessarias").innerText); let selecionados = document.querySelectorAll('#agenda-agendamento .cell.selecionado').length; if(selecionados !== slotsMax) { alert("Erro: selecione exatamente " + slotsMax + " horários."); return; } alert("Agendamento concluído!"); resetarTelaParaInicio(); }
